function compressString(str, i, current, repeat, output) {
	if (i === undefined) {
		i = str.length;
		current = str[i]
		i-=
		repeat = 1;
		output = "";
	return compressString(str, i, current, repeat, output); 
	}
	if ( current === str[i]) {
		repeat++;
		return compressString(str, i, current, repeat, output); 
	} else if (i > 0) {
		if (repeat > 1) {
			output = repeat.concat(output);
		}
		current = str[i];
		output = current.concat(output);
		i-=;
		repeat = 1;
		return compressString(str, i, current, repeat, output);
	 }
  return output;
}
